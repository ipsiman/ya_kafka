# Создание топика из консоли
docker exec -it unit_1_final-kafka-0-1 kafka-topics.sh --create \
  --topic final-1-topic \
  --bootstrap-server localhost:9092 \
  --partitions 3 \
  --replication-factor 2

# Описание топика
docker exec -it unit_1_final-kafka-0-1 kafka-topics.sh \
  --describe \
  --topic final-1-topic \
  --bootstrap-server localhost:9092

Topic: final-1-topic    TopicId: m2Xlh3DaSFSkVhgTqSL3_g PartitionCount: 3       ReplicationFactor: 2    Configs:
        Topic: final-1-topic    Partition: 0    Leader: 0       Replicas: 0,1   Isr: 0,1
        Topic: final-1-topic    Partition: 1    Leader: 1       Replicas: 1,2   Isr: 1,2
        Topic: final-1-topic    Partition: 2    Leader: 2       Replicas: 2,0   Isr: 2,0

# Удаление топика
docker exec -it unit_1_final-kafka-0-1 kafka-topics.sh \
  --delete \
  --topic final-1-topic \
  --bootstrap-server localhost:9092

# Список топиков
docker exec -it unit_1_final-kafka-0-1 kafka-topics.sh \
  --list \
  --bootstrap-server localhost:9092
